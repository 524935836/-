function debounce(fn, wait, immediate) {
  let timer, res
  const debounced = function (...args) {
    if(timer) clearTimeout(timer)
    if(immediate) {
      let callNew = !timer
      timer = setTimeout(() => {
        timer = null
        if(!callNew) {
        res = fn.apply(this, args)
        }
      }, wait)
      // 判断是否立即执行
      if(callNew) {
        res = fn.apply(this, args)
      } 
    } else {
      timer = setTimeout(() => {
        res = fn.apply(this, args)
      }, wait)
    }
    return res
  }

  debounced.cancel = function () {
    clearTimeout(timer)
    timer = null
  }
  return debounced
}
